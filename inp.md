  ORUVELA (KANNA_n == 0){
    ITHU EPDI IRUKU 0:
  }
  ORUVELA (KANNA_n == 1){
    ITHU EPDI IRUKU 1:
  }
  ITHU EPDI IRUKU KANNA_fib(KANNA_n-1) + KANNA_fib(KANNA_n-2):
}
